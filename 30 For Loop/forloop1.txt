Example Script

@echo off
setlocal enabledelayedexpansion
goto :main

:main
setlocal

	set string=Hello World^^!
	echo !string!
	echo.

	:: set /a value=0

	for %%g in ( 1 2 3 4 5 ) do (
		echo %%g
		if !data! equ %%g (
			echo Data !data! - Range %%g
		)
	)

endlocal
goto :eof


NOTED :
EQU = equal
NEQ = not Equal
LSS = less than
LEQ = less than or equal
GTR = greater than
GEQ = greater than or equal
