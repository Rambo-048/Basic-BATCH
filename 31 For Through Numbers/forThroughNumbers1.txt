FOR THROUGH NUMBERS

Example Script :
----------------------------------------------------------
Example 1
@echo off
setlocal enabledelayedexpansion
goto :main

:main
setlocal

	for %%g in ( This is set function ) do (
		echo %%g
	)	

endlocal
goto :eof
