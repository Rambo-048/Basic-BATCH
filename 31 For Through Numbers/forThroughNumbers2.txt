FOR THROUGH NUMBERS

Example Script :
----------------------------------------------------------
Example 1
@echo off
setlocal enabledelayedexpansion
goto :main

:main
setlocal

	for %%g in ( Sometimes I Want To Treat People How They Treat Me, But I Cant Because Thats Not My Character ) do (
		echo %%g
	)	

endlocal
goto :eof
