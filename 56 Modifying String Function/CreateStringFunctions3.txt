CREATE STRING FUNCTIONS

EXAMPLE SCRIPT
------------------------------------------------------------
Example 1

@echo off

set string=%2
set /a test=5

for /f "useback tokens=*" %%a in ('%string%') do (
	
	if %string% equ %%~a (
		set /a test=3
	)	

	set string=%%~a
)

echo %2 > %TEMP%\tempfile.txt

for %%g in (%TEMP%\tempfile.txt) do (
	set /a %1_length=%%~zg - %test%
)

del %TEMP%\tempfile.txt
set %1=%string%
--------------------------------------------------------------

Example 2

@echo off

set string=%2


for /f "useback tokens=*" %%a in ('%string%') do (
	
	set string=%%~a
)

echo %2 > %TEMP%\tempfile.txt

for %%g in (%TEMP%\tempfile.txt) do (
	set /a %1_length=%%~zg - 5
)

del %TEMP%\tempfile.txt
set %1=%string%
